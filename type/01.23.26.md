
// tipos de datos

// boolean que son los lógicos, ya los conoces
let logico: boolean = false // se puede visualizar que infirio que es un boolean, esto es por typescript
let verdadero: boolean = true

// number, incluye enteros y decimales en su lógica
// tambien puede incluir hexadecimales
let entero: number = 44
let decimal: number  = 26.45
let hexadecimales: number = 0x01234a
let octal: number = 0o700 // para el octal necesitas agregar un 'o' despues del primer 0
let largenumber: number = 3_500_322 // como puedes ver puedes usar el '_' para separar

// String
let cadenatexto: string = 'hola mundo \n ¿como estas?'
console.log(cadenatexto)

// los siguientes tipos no estan en javascript

// any, representa cualquier cosa y en teoría no se deberia de utilizar casi nunca.
// pero si se usa es para definir que no le de reglas de ningun tipo
// o sea decirle a tp que no es algun tipo en especifico.

let cualquiera: any
cualquiera = false
cualquiera = "hola"
cualquiera = 123

// otro tipo de datos es el null, simplemente es nada, rara vez se usa

let nada: null = null
console.log(null)

// undefined, como null es practicamente nada, talvez se use a futuro
let indefinido: undefined = undefined

// void para indicar que una función no devuelve nada (el dafult de una funcion tambien)
// normalemente es un tipo para funciones
function saludar(): void { // si dejas el mouse encima te daras cuenta que es void la funcion
    console.log('hola')
}
